CREATE TABLE `line` (
  `id` int(11) NOT NULL AUTO_INCREMENT COMMENT '自增主键',
  `line_name` varchar(64) NOT NULL COMMENT '业务线名称',
  `line_no` varchar(32) NOT NULL COMMENT '业务线编码',
  `url` varchar(128) NOT NULL DEFAULT '' COMMENT 'core地址url',
  `is_valid` tinyint(1) NOT NULL COMMENT '是否有效',
  `is_delete` tinyint(1) NOT NULL COMMENT '是否删除',
  `create_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `update_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `line_no` (`line_no`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='业务线表';

CREATE TABLE `rule` (
  `id` int(11) NOT NULL AUTO_INCREMENT COMMENT '自增主键',
  `line_no` varchar(32) NOT NULL COMMENT '业务线编码',
  `rule_name` varchar(32) NOT NULL COMMENT '规则名称',
  `rule_no` varchar(32) NOT NULL COMMENT '规则编码',
  `data_source_no` varchar(32) NOT NULL COMMENT '数据源编码',
  `cond` varchar(32) NOT NULL COMMENT '条件',
  `threshold` varchar(256) NOT NULL COMMENT '阈值',
  `threshold_type` varchar(256) NOT NULL COMMENT '阈值类型',
  `result` varchar(16) NOT NULL COMMENT '结果',
  `version` int(11) NOT NULL COMMENT '版本',
  `deploy_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '部署时间',
  `is_valid` tinyint(1) NOT NULL COMMENT '是否有效',
  `is_delete` tinyint(1) NOT NULL COMMENT '是否删除',
  `create_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `update_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `rule_no` (`rule_no`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='规则表';
